public class HashTable
{  
  private int size;
  private ListNode[] table;

  public HashTable(int numSlots)
  {
    size = numSlots;
    table = new ListNode[size];
  }

  public void add(Object obj)
  {
    ListNode element;
    element = new ListNode(obj, null);

    int index = obj.hashCode() % size;

    if(table[index] == null)
      table[index] = element;
    else{
      ListNode current = table[index];
      while(current.getNext() != null)
        current = current.getNext();
      current.setNext(element);
    }
  }
}

class HashTable
{  
  private LinkedList[] table;
     
  public HashTable(int size) 
  {
      table = new LinkedList[size];
  }
     
  public void add(Object obj) 
  {
    int index = obj.hashCode() % table.length;
    if(table[index] == null)                      
        table[index] = new LinkedList();
    table[index].add(obj);
  }
     
  public String toString()
  {
    String out = "";
      for( LinkedList ll : table)
         out += ll + "\n";
      return out;
  }
}

public class HashTable
{  
  private int size;
  private int[] aray;

  public HashTable(int numSlots)
  {
    size = numSlots;
    aray = new int[size];
  }

  public void add(int num)
  {
    int index = num % size;
    if(aray[index] == 0)
      aray[index] = num;
    else
    {
      index++;
      while(aray[index] != 0)
        index++;
      aray[index] = num;
     }
  }

  public String toString()
  {
    String out = "";
    out += Arrays.toString( aray );
    return out;
  }
}

  h.add(v);
